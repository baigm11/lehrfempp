166  SCALAR, DIFF_COEFF, REACTION_COEFF>::ElemMat const
167 ReactionDiffusionElementMatrixProvider<
168  SCALAR, DIFF_COEFF, REACTION_COEFF>::Eval(const lf::mesh::Entity &cell) {
169  // Topological type of the cell
170  const lf::base::RefEl ref_el{cell.RefEl()};
171  PrecomputedScalarReferenceFiniteElement<SCALAR> &pfe =
172  fe_precomp_[ref_el.Id()];
173 
174  // Query the shape of the cell
175  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
176  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
177  LF_ASSERT_MSG((geo_ptr->DimLocal() == 2),
178  "Only 2D implementation available!");
179  SWITCHEDSTATEMENT(ctrl_, kout_cell,
180  std::cout << ref_el << ", shape = \n"
181  << geo_ptr->Global(ref_el.NodeCoords())
182  << std::endl);
183  // Physical dimension of the cell
184  const dim_t world_dim = geo_ptr->DimGlobal();
185 
186  const Eigen::VectorXd determinants(
187  geo_ptr->IntegrationElement(pfe.Qr().Points()));
188  LF_ASSERT_MSG(
189  determinants.size() == pfe.Qr().NumPoints(),
190  "Mismatch " << determinants.size() << " <-> " << pfe.Qr().NumPoints());
191  // Fetch the transformation matrices for the gradients
192  const Eigen::MatrixXd JinvT(
193  geo_ptr->JacobianInverseGramian(pfe.Qr().Points()));
194  LF_ASSERT_MSG(
195  JinvT.cols() == 2 * pfe.Qr().NumPoints(),
196  "Mismatch " << JinvT.cols() << " <-> " << 2 * pfe.Qr().NumPoints());
197  LF_ASSERT_MSG(JinvT.rows() == world_dim,
198  "Mismatch " << JinvT.rows() << " <-> " << world_dim);
199 
200  // compute values of alpha, gamma at quadrature points:
201  auto alphaval = alpha_(cell, pfe.Qr().Points());
202  auto gammaval = gamma_(cell, pfe.Qr().Points());
203 
204  // Element matrix
205  elem_mat_t mat(pfe.NumRefShapeFunctions(), pfe.NumRefShapeFunctions());
206  mat.setZero();
207 
208  // Loop over quadrature points
209  for (int k = 0; k < pfe.Qr().NumPoints(); ++k) {
210  const double w = pfe.Qr().Weights()[k] * determinants[k];
211  // Transformed gradients
212  const auto trf_grad(JinvT.block(0, 2 * k, world_dim, 2) *
213  pfe.PrecompGradientsReferenceShapeFunctions()
214  .block(0, 2 * k, mat.rows(), 2)
215  .transpose());
216  // Transformed gradients multiplied with coefficient
217  const auto alpha_trf_grad(alphaval[k] * trf_grad);
218  mat += w * (alpha_trf_grad.transpose() * trf_grad +
219  (gammaval[k] * pfe.PrecompReferenceShapeFunctions().col(k)) *
220  (pfe.PrecompReferenceShapeFunctions().col(k).transpose()));
221  }
222  return mat;
223 }
