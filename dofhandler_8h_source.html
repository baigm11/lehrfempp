425  template <typename LOCALDOFINFO>
426  DynamicFEDofHandler(std::shared_ptr<const lf::mesh::Mesh> mesh_p,
427  LOCALDOFINFO &&locdof)
428  : mesh_p_(std::move(mesh_p)) {
429  LF_ASSERT_MSG((mesh_p_->DimMesh() == 2), "Can handle 2D meshes only");
430 
431  // Index counter for global shape functions = global dof
432  gdof_idx_t dof_idx = 0;
433 
434  // Step I: Set indices for shape functions on nodes
435  // Run through node indices (entities of co-dimension 2)
436  const size_type no_nodes = mesh_p_->NumEntities(2);
437  no_int_dofs_[2].resize(no_nodes, 0);
438  offsets_[2].resize(no_nodes + 1, 0);
439  // Traverse nodes (co-dimension-2 entities) based on indices
440  for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
441  // Obtain pointer to node entity
442  const mesh::Entity *node_p{mesh_p_->EntityByIndex(2, node_idx)};
443  LF_ASSERT_MSG(mesh_p_->Index(*node_p) == node_idx, "Node index mismatch");
444  // Offset for indices of node in index vector
445  glb_idx_t node_dof_offset = dof_idx;
446  offsets_[2][node_idx] = node_dof_offset;
447  // Request number of local shape functions associated with the node
448  size_type no_int_dof_node = locdof(*node_p);
449  no_int_dofs_[2][node_idx] = no_int_dof_node;
450 
451  // Store dof indices in array
452  for (int j = 0; j < no_int_dof_node; j++) {
453  dofs_[2].push_back(dof_idx);
454  dof_entities_.push_back(node_p); // Store entity for current dof
455  dof_idx++; // Move on to next index
456  }
457  }
458  // Set sentinel
459  offsets_[2][no_nodes] = dof_idx;
460 
461  // Step II: Set indices for shape functions on edges (co-dimension = 1)
462  const size_type no_edges = mesh_p_->NumEntities(1);
463  // Set length of edge-related index vectors
464  no_int_dofs_[1].resize(no_edges, 0);
465  offsets_[1].resize(no_edges + 1, 0);
466  // points to the position of the current dof index
467  size_type edge_dof_offset = 0;
468  for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
469  // Obtain pointer to edge entity
470  const mesh::Entity *edge{mesh_p_->EntityByIndex(1, edge_idx)};
471  LF_ASSERT_MSG(mesh_p_->Index(*edge) == edge_idx, "Edge index mismatch");
472  // Offset for indices of edge dof in index vector
473  offsets_[1][edge_idx] = edge_dof_offset;
474  size_type no_int_dof_edge = locdof(*edge);
475  no_int_dofs_[1][edge_idx] = no_int_dof_edge;
476 
477  // Obtain indices for basis functions sitting at endpoints
478  // Endpoints are mesh entities with co-dimension = 2
479  for (const lf::mesh::Entity &endpoint : edge->SubEntities(1)) {
480  const glb_idx_t ep_idx(mesh_p_->Index(endpoint));
481  glb_idx_t ep_dof_offset = offsets_[2][ep_idx];
482  size_type no_int_dofs_ep = no_int_dofs_[2][ep_idx];
483  // Copy indices of shape functions from nodes to edge
484  for (int j = 0; j < no_int_dofs_ep; j++) {
485  dofs_[1].push_back(dofs_[2][ep_dof_offset + j]);
486  edge_dof_offset++;
487  }
488  }
489  // Set indices for interior edge degrees of freedom
490  for (int j = 0; j < no_int_dof_edge; j++) {
491  dofs_[1].push_back(dof_idx);
492  edge_dof_offset++;
493  dof_entities_.push_back(edge);
494  dof_idx++;
495  }
496  }
497  // Set sentinel
498  offsets_[1][no_edges] = edge_dof_offset;
499 
500  // Step III: Set indices for shape functions on cells (co-dimension = 0)
501  const size_type no_cells = mesh_p_->NumEntities(0);
502  // Set length of cell-related index vectors
503  no_int_dofs_[0].resize(no_cells, 0);
504  offsets_[0].resize(no_cells + 1, 0);
505  // points to the position of the current dof index
506  size_type cell_dof_offset = 0;
507  for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
508  // Obtain pointer to current ell
509  const mesh::Entity *cell{mesh_p_->EntityByIndex(0, cell_idx)};
510  // Offset for indices of cell dof in index vector
511  offsets_[0][cell_idx] = cell_dof_offset;
512  size_type no_int_dof_cell = locdof(*cell);
513  no_int_dofs_[0][cell_idx] = no_int_dof_cell;
514 
515  // Obtain indices for basis functions in vertices
516  for (const lf::mesh::Entity &vertex : cell->SubEntities(2)) {
517  const glb_idx_t vt_idx(mesh_p_->Index(vertex));
518  glb_idx_t vt_dof_offset = offsets_[2][vt_idx];
519  size_type no_int_dofs_vt = no_int_dofs_[2][vt_idx];
520  // Copy indices of shape functions from nodes to cell
521  for (int j = 0; j < no_int_dofs_vt; j++) {
522  dofs_[0].push_back(dofs_[2][vt_dof_offset + j]);
523  cell_dof_offset++;
524  }
525  }
526 
527  // Collect indices of interior shape functions of edges
528  // Internal ordering will depend on the orientation of the edge
529  lf::base::RandomAccessRange<const lf::mesh::Orientation>
530  edge_orientations(cell->RelativeOrientations());
531  lf::base::RandomAccessRange<const lf::mesh::Entity> edges(
532  cell->SubEntities(1));
533  // Loop over edges
534  for (int ed_sub_idx = 0; ed_sub_idx < cell->RefEl().NumSubEntities(1);
535  ed_sub_idx++) {
536  const glb_idx_t edge_idx = mesh_p_->Index(edges[ed_sub_idx]);
537  const size_type no_int_dof_edge = no_int_dofs_[1][edge_idx];
538  const glb_idx_t edge_int_dof_offset =
539  offsets_[1][edge_idx + 1] - no_int_dof_edge;
540 
541  // Copy indices of shape functions from edges to cell
542  // The order, in which they are copied depends on the relative
543  // orientation of the edge w.r.t. the cell
544  switch (edge_orientations[ed_sub_idx]) {
545  case lf::mesh::Orientation::positive: {
546  for (int j = 0; j < no_int_dof_edge; j++) {
547  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
548  cell_dof_offset++;
549  }
550  break;
551  }
552  case lf::mesh::Orientation::negative: {
553  for (int j = no_int_dof_edge - 1; j >= 0; j--) {
554  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
555  cell_dof_offset++;
556  }
557  break;
558  }
559  } // end switch
560  } // end loop over edges
561 
562  // Set indices for interior shape functions of the cell
563  for (int j = 0; j < no_int_dof_cell; j++) {
564  dofs_[0].push_back(dof_idx);
565  cell_dof_offset++;
566  dof_entities_.push_back(cell);
567  dof_idx++;
568  } // end loop over interior dof of cell
569  } // end loop over cells
570  // Set sentinel
571  offsets_[0][no_cells] = cell_dof_offset;
572 
573  // Finally set number of global shape functions
574  num_dof_ = dof_idx;
575  } // end constructor
