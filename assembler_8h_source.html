94 void AssembleMatrixLocally(dim_t codim, const DofHandler &dof_handler_trial,
95  const DofHandler &dof_handler_test,
96  ENTITY_MATRIX_PROVIDER &entity_matrix_provider,
97  TMPMATRIX &matrix) {
98  // Pointer to underlying mesh
99  auto mesh = dof_handler_trial.Mesh();
100 
101  LF_ASSERT_MSG(mesh == dof_handler_test.Mesh(),
102  "Trial and test space must be defined on the same mesh");
103 
104  // Central assembly loop over entities of co-dimension specified by
105  // the template argument CODIM
106  for (const lf::mesh::Entity &entity : mesh->Entities(codim)) {
107  // Some entities may be skipped
108  if (entity_matrix_provider.isActive(entity)) {
109  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl, amd_entity,
110  std::cout << "ASM: " << entity << '('
111  << mesh->Index(entity) << ')' << std::endl);
112  // Size, aka number of rows and columns, of element matrix
113  const size_type nrows_loc = dof_handler_test.NoLocalDofs(entity);
114  const size_type ncols_loc = dof_handler_trial.NoLocalDofs(entity);
115  // row indices of for contributions of cells
116  lf::base::RandomAccessRange<const gdof_idx_t> row_idx(
117  dof_handler_test.GlobalDofIndices(entity));
118  // Column indices of for contributions of cells
119  lf::base::RandomAccessRange<const gdof_idx_t> col_idx(
120  dof_handler_trial.GlobalDofIndices(entity));
121  // Request local matrix from entity_matrix_provider object. In the case
122  // codim = 0, when `entity` is a cell, this is the element matrix
123  const auto elem_mat{entity_matrix_provider.Eval(entity)};
124  LF_ASSERT_MSG(elem_mat.rows() >= nrows_loc,
125  "nrows mismatch " << elem_mat.rows() << " <-> " << nrows_loc
126  << ", entity " << mesh->Index(entity));
127  LF_ASSERT_MSG(elem_mat.cols() >= ncols_loc,
128  "ncols mismatch " << elem_mat.cols() << " <-> " << nrows_loc
129  << ", entity " << mesh->Index(entity));
130  // clang-format off
131  SWITCHEDSTATEMENT(
132  ass_mat_dbg_ctrl, amd_gdof,
133  std::cout << "ASM: row_idx = ";
134  for (auto gdof_idx: row_idx) {
135  std::cout << gdof_idx << ' ';
136  }
137  std::cout << std::endl << "ASM: col_idx = ";
138  for (auto gdof_idx : col_idx) {
139  std::cout << gdof_idx << ' ';
140  }
141  std::cout << std::endl);
142  // clang-format on
143  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl, amd_lmdim,
144  std::cout << "ASM: " << nrows_loc << " x " << ncols_loc
145  << " element matrix" << std::endl);
146  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl, amd_locmat,
147  for (int i = 0; i < nrows_loc; i++) {
148  std::cout << "[ ";
149  for (int j = 0; j < ncols_loc; j++) {
150  std::cout << elem_mat(i, j) << ' ';
151  }
152  std::cout << "]" << std::endl;
153  });
154  // Assembly double loop
155  for (int i = 0; i < nrows_loc; i++) {
156  for (int j = 0; j < ncols_loc; j++) {
157  // Add the element at position (i,j) of the local matrix
158  // to the entry at (row_idx[i], col_idx[j]) of the global matrix
159  matrix.AddToEntry(row_idx[i], col_idx[j], elem_mat(i, j));
160  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl,amd_lass ,
161  std::cout << "(" << row_idx[i] << ','
162  << col_idx[j] << ")+= " << elem_mat(i, j) << ", ";
163  );
164  }
165  } // end assembly local double loop
166  SWITCHEDSTATEMENT(ass_mat_dbg_ctrl,amd_lass ,
167  std::cout << std::endl;
168  );
169  } // end if(isActive() )
170  } // end main assembly loop
171 } // end AssembleMatrixLocally

244 template <typename VECTOR, class ENTITY_VECTOR_PROVIDER>
245 void AssembleVectorLocally(dim_t codim, const DofHandler &dof_handler,
246  ENTITY_VECTOR_PROVIDER &entity_vector_provider, VECTOR &resultvector) {
247  // Pointer to underlying mesh
248  auto mesh = dof_handler.Mesh();
249 
250  // Central assembly loop over entities of the co-dimension specified via
251  // the template argument CODIM
252  for (const lf::mesh::Entity &entity : mesh->Entities(codim)) {
253  // Some cells may be skipped
254  if (entity_vector_provider.isActive(entity)) {
255  // Length of element vector
256  const size_type veclen = dof_handler.NoLocalDofs(entity);
257  // global dof indices for contribution of the entity
258  lf::base::RandomAccessRange<const gdof_idx_t> dof_idx(
259  dof_handler.GlobalDofIndices(entity));
260  // Request local vector from entity_vector_provider object. In the case CODIM = 0,
261  // when `entity` is a cell, this is the element vector
262  const auto elem_vec{entity_vector_provider.Eval(entity)};
263  LF_ASSERT_MSG(elem_vec.size() >= veclen,
264  "length mismatch " << elem_vec.size() << " <-> " << veclen
265  << ", entity " << mesh->Index(entity));
266  // Assembly (single) loop
267  for (int i = 0; i < veclen; i++) {
268  resultvector[dof_idx[i]] += elem_vec[i];
269  } // end assembly localloop
270  } // end if(isActive() )
271  } // end main assembly loop
272 } // end AssembleVectorLocally
