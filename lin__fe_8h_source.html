157  const lf::base::size_type num_nodes{ref_el.NumNodes()};
158 
159  // Obtain the vertex coordinates of the cell, which completely
160  // describe its shape.
161  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
162  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
163  LF_ASSERT_MSG((geo_ptr->DimGlobal() == 2) && (geo_ptr->DimLocal() == 2),
164  "Only 2D implementation available!");
165  const Eigen::MatrixXd &ref_el_corners(ref_el.NodeCoords());
166  // World coordinates of vertices
167  // const Eigen::MatrixXd vertices{geo_ptr->Global(ref_el_corners)};
168  // Debugging output
169  SWITCHEDSTATEMENT(dbg_ctrl, dbg_geo,
170  std::cout << ref_el << ", shape = \n"
171  << (geo_ptr->Global(ref_el_corners))
172  << std::endl);
173  // Midpoints of edges in the reference cell
174  Eigen::MatrixXd ref_mp(2, 4);
175  switch (ref_el) {
176  case lf::base::RefEl::kTria(): {
177  // clang-format off
178  ref_mp << 0.5,0.5,0.0,-1.0,
179  0.0,0.5,0.5,-1.0;
180  // clang-format on
181  break;
182  }
183  case lf::base::RefEl::kQuad(): {
184  // clang-format off
185  ref_mp << 0.5,1,0.5,0.0,
186  0.0,0.5,1.0,0.5;
187  // clang-format on
188  break;
189  }
190  default: {
191  LF_ASSERT_MSG(false, "Illegal entity type!");
192  break;
193  }
194  } // end switch
195 
196  const double area = lf::geometry::Volume(*geo_ptr);
197 
198  // Vector for returning element vector
199  elem_vec_t elem_vec = elem_vec_t::Zero();
200  // Run over the midpoints of edges and fetch values of the source function
201  // there
202 
203  auto fvals = f_(cell, ref_mp);
204 
205  for (int k = 0; k < num_nodes; k++) {
206  elem_vec[k] += 0.5 * fvals[k];
207  elem_vec[(k + 1) % num_nodes] += 0.5 * fvals[k];
208  }
209  SWITCHEDSTATEMENT(
210  dbg_ctrl, dbg_locvec,
211  std::cout << "element vector = " << elem_vec.head(num_nodes).transpose()
212  << std::endl);
213  return (area / num_nodes) * elem_vec;
214 }
