421  bool operator()(const lf::mesh::Entity &ent) const {
422  const lf::base::RefEl ref_el_type = ent.RefEl();
423  // Obtain shape of entity
424  lf::geometry::Geometry *geo_ptr = ent.Geometry();
425  LF_ASSERT_MSG(geo_ptr != nullptr, "Missing geometry for " << ent);
426  switch (ref_el_type) {
427  case lf::base::RefEl::kPoint(): {
428  Eigen::MatrixXd pos(geo_ptr->Global(kpoint_center_));
429  return pos_pred_(pos);
430  }
431  case lf::base::RefEl::kSegment(): {
432  Eigen::MatrixXd pos(geo_ptr->Global(kedge_center_));
433  return pos_pred_(pos);
434  }
435  case lf::base::RefEl::kTria(): {
436  Eigen::MatrixXd pos(geo_ptr->Global(ktria_center_));
437  return pos_pred_(pos);
438  }
439  case lf::base::RefEl::kQuad(): {
440  Eigen::MatrixXd pos(geo_ptr->Global(kquad_center_));
441  return pos_pred_(pos);
442  }
443  default: {
444  LF_ASSERT_MSG(false, "Illegal entity type");
445  break;
446  }
447  } // end switch
448  return false;
449  }
