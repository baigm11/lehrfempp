56 template <typename LOC_COMP, typename COEFFVECTOR, typename SELECTOR>
57 double SumCellFEContrib(const lf::assemble::DofHandler &dofh,
58  LOC_COMP &loc_comp, const COEFFVECTOR &uh,
59  SELECTOR &&pred) {
60  // Type for passing local coefficient vectors
61  using dofvector_t = std::vector<typename COEFFVECTOR::value_type>;
62  // Retrieve the mesh
63  const lf::mesh::Mesh &mesh{*dofh.Mesh()};
64 
65  // Check whether sufficiently large vector uh
66  LF_ASSERT_MSG(uh.size() >= dofh.NoDofs(), "uh vector too short!");
67 
68  double sum = 0.0;
70  for (const lf::mesh::Entity &cell : mesh.Entities(0)) {
71  if (pred(cell)) {
72  // Number of local shape functions
73  const size_type num_loc_dofs = dofh.NoLocalDofs(cell);
74  // Allocate a temporary vector of appropriate size
75  dofvector_t loc_coeffs(num_loc_dofs);
76  // Fetch global numbers of local shape functions
77  lf::base::RandomAccessRange<const gdof_idx_t> ldof_gidx(
78  dofh.GlobalDofIndices(cell));
79  // Copy degrees of freedom into temporory vector
80  for (int j = 0; j < num_loc_dofs; ++j) {
81  loc_coeffs[j] = uh[ldof_gidx[j]];
82  }
83  // Sum local contribution
84  sum += loc_comp(cell, loc_coeffs);
85  } // end if (isACtive())
86  }
87  return sum;
88 }

186 auto NodalProjection(std::shared_ptr<ScalarUniformFESpace<SCALAR>> fe_space,
187  FUNCTOR &&u, SELECTOR &&pred = base::PredicateTrue{}) {
188  static_assert(isMeshFunction<std::remove_reference_t<FUNCTOR>>);
189  // choose scalar type so it can hold the scalar type of u as well as SCALAR
190  using scalarMF_t = MeshFunctionReturnType<std::remove_reference_t<FUNCTOR>>;
191  using scalar_t = decltype(SCALAR(0) * scalarMF_t(0));
192  // Return type, type for FE coefficient vector
193  using dof_vec_t = Eigen::Matrix<scalar_t, Eigen::Dynamic, 1>;
194 
195  // Underlying mesh instance
196  const lf::mesh::Mesh &mesh{*fe_space->Mesh()};
197  // Fetch local-to-global index mapping for shape functions
198  const lf::assemble::DofHandler &dofh{fe_space->LocGlobMap()};
199 
200  // Vector for returning expansion coefficients
201  dof_vec_t glob_dofvec(dofh.NoDofs());
202  glob_dofvec.setZero();
203 
204  // Loop over all cells
205  for (const lf::mesh::Entity &cell : mesh.Entities(0)) {
206  if (!pred(cell)) {
207  continue;
208  }
209  // Topological type of the cell
210  const lf::base::RefEl ref_el{cell.RefEl()};
211 
212  // TODO(ralfh) uncommend when ctrl_prj is well-defined.
213  // SWITCHEDSTATEMENT(ctrl_prj, kout_prj_cell,
214  // std::cout << ref_el << ", shape = \n"
215  // << geo_ptr->Global(ref_el.NodeCoords())
216  // << std::endl);
217 
218  // Information about local shape functions on reference element
219  const ScalarReferenceFiniteElement<double> &ref_shape_fns{
220  *fe_space->ShapeFunctionLayout(ref_el)};
221  // Number of evaluation nodes
222  const size_type num_eval_nodes = ref_shape_fns.NumEvaluationNodes();
223  // Obtain reference coordinates for evaluation nodes
224  const Eigen::MatrixXd ref_nodes(ref_shape_fns.EvaluationNodes());
225 
226  // Collect values of function to be projected in a row vector
227  auto uvalvec = u(cell, ref_nodes);
228 
229  // Compute the resulting local degrees of freedom
230  auto dofvec(ref_shape_fns.NodalValuesToDofs(
231  Eigen::Map<Eigen::Matrix<scalar_t, Eigen::Dynamic, 1>>(
232  &uvalvec[0], uvalvec.size(), 1)));
233 
234  // Set the corresponing global degrees of freedom
235  // Note: "Setting", not "adding to"
236  // Number of local shape functions
237  const size_type num_loc_dofs = dofh.NoLocalDofs(cell);
238  LF_ASSERT_MSG(
239  dofvec.size() == num_loc_dofs,
240  "Size mismatch: " << dofvec.size() << " <-> " << num_loc_dofs);
241  // Fetch global numbers of local shape functions
242  lf::base::RandomAccessRange<const gdof_idx_t> ldof_gidx(
243  dofh.GlobalDofIndices(cell));
244  // Insert dof values into the global coefficient vector
245  for (int j = 0; j < num_loc_dofs; ++j) {
246  glob_dofvec[ldof_gidx[j]] = dofvec[j];
247  }
248  }
249  return glob_dofvec;
250 }

289 template <typename SCALAR, typename EDGESELECTOR, typename FUNCTION>
290 std::vector<std::pair<bool, SCALAR>> InitEssentialConditionFromFunction(
291  const lf::assemble::DofHandler &dofh,
292  const ScalarReferenceFiniteElement<SCALAR> &fe_spec_edge,
293  EDGESELECTOR &&esscondflag, FUNCTION &&g) {
294  static_assert(isMeshFunction<std::remove_reference_t<FUNCTION>>);
295  // static_assert(isMeshFunction<FUNCTION>, "g must by a MeshFunction object");
296  LF_ASSERT_MSG(fe_spec_edge.RefEl() == lf::base::RefEl::kSegment(),
297  "finite element specification must be for an edge!");
298 
299  // *** I: Preprocessing ***
300  // Fetch numbers of evaluation nodes and local shape functions
301  const size_type num_eval_pts = fe_spec_edge.NumEvaluationNodes();
302  const size_type num_rsf = fe_spec_edge.NumRefShapeFunctions();
303 
304  // Preprocessing: obtain evaluation nodes on reference segment [0,1]
305  const Eigen::MatrixXd ref_eval_pts{fe_spec_edge.EvaluationNodes()};
306  Eigen::MatrixXd eval_pts(2, num_eval_pts);
307  Eigen::Matrix<SCALAR, 1, Eigen::Dynamic> dof_vals(num_eval_pts);
308 
309  // Underlying mesh
310  const lf::mesh::Mesh &mesh{*dofh.Mesh()};
311 
312  // Vector for returning flags and fixed values for dofs
313  const size_type num_coeffs = dofh.NoDofs();
314  std::vector<std::pair<bool, SCALAR>> flag_val_vec(num_coeffs,
315  {false, SCALAR{}});
316 
317  // *** II: Local computations ****
318  // Visit all edges of the mesh (codim-1 entities)
319  for (const lf::mesh::Entity &edge : mesh.Entities(1)) {
320  // Check whether the current edge carries dofs to be imposed by the
321  // function g. The decision relies on the predicate `esscondflag`
322  if (esscondflag(edge)) {
323  // Fetch the shape of the edge
324  const lf::geometry::Geometry *edge_geo_p{edge.Geometry()};
325  // Evaluate mesh function at several points specified by their
326  // reference coordinates.
327  auto g_vals = g(edge, ref_eval_pts);
328 
329  // Compute degrees of freedom from function values in evaluation points
330  dof_vals = fe_spec_edge.NodalValuesToDofs(
331  Eigen::Map<Eigen::Matrix<SCALAR, 1, Eigen::Dynamic>>(&g_vals[0], 1,
332  g_vals.size()));
333  LF_ASSERT_MSG(dof_vals.size() == num_rsf,
334  "Mismatch " << dof_vals.size() << " <-> " << num_rsf);
335  LF_ASSERT_MSG(
336  dofh.NoLocalDofs(edge) == num_rsf,
337  "Mismatch " << dofh.NoLocalDofs(edge) << " <-> " << num_rsf);
338  // Fetch indices of global shape functions associated with current edge
339  auto gdof_indices{dofh.GlobalDofIndices(edge)};
340  int k = 0;
341  // Set flags and values; setting, no accumulation here!
342  for (const lf::assemble::gdof_idx_t gdof_idx : gdof_indices) {
343  flag_val_vec[gdof_idx] = {true, dof_vals[k++]};
344  }
345  }
346  }
347  return flag_val_vec;
