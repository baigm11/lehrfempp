143  "Only 2D implementation available!");

145  "Dof vector too short");

147  const Eigen::MatrixXd &qr_Points{fe.Qr().Points()};
148  const Eigen::VectorXd &qr_Weights{fe.Qr().Weights()};
149  const size_type qr_NumPts = fe.Qr().NumPoints();
150  const size_type num_LSF = fe.NumRefShapeFunctions();
151  const auto &rsf_QuadPts{fe.PrecompReferenceShapeFunctions()};
152

154  std::cout << ref_el << ", (Nlsf = " << num_LSF
155  << ", Nqr = " << qr_NumPts << ") shape = \n"
156  << geo_ptr->Global(ref_el.NodeCoords())
157  << std::endl);
158 
159  // World coordinates of quadrature points
160  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(qr_Points));
161  // Obtain the metric factors for the quadrature points
162  const Eigen::VectorXd determinants(geo_ptr->IntegrationElement(qr_Points));
163  SWITCHEDSTATEMENT(
164  ctrl_, kout_comp, std::cout << "\t dofs = "; for (auto x
165  : dofs) {

167  } std::cout << "\t dets "
168  << determinants.transpose() << std::endl);
169 
170  double sum = 0.0; // summation variable
171 
172  auto uvals = u_(cell, qr_Points);
173 
174  // Loop over quadrature points
175  for (int k = 0; k < qr_NumPts; ++k) {
176  // Value of finite element function at quadrature point
177  // Loop over local shape functions/dofs to compute the value of
